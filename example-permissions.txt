# Permissions: donateteleport.use
# Can use: /dtp spawn

# VIP group - donors with VIP rank
# Permissions: donateteleport.use, donateteleport.vip
# Can use: /dtp spawn, /dtp home, /dtp vip

# Premium group - donors with Premium rank
# Permissions: donateteleport.use, donateteleport.vip, donateteleport.premium
# Can use: All commands including /dtp premium

# Admin group - server administrators
# Permissions: donateteleport.use, donateteleport.vip, donateteleport.premium, donateteleport.admin
# Can use: All commands including /dtpreload and /dtpadmin

# /lp group default permission set donateteleport.use true

# /lp group vip permission set donateteleport.vip true

# /lp group premium permission set donateteleport.premium true
